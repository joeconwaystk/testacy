    : super(config);

Compose modular, maintainable code.

import 'package:aqueduct/aqueduct.dart';


class PersonController extends HTTPController {

  @httpGet

  Future<Response> getAll(

      {@HTTPQuery("name") String name}) async {

    var people = await fetchPeople(byName: name);

    return new Response.ok(people);

  }


  @httpGet

  Future<Response> getOne(

      @HTTPPath("id") int id) async {

    var person = await fetchPerson(id);

    return new Response.ok(person);

  }

}

import 'package:aqueduct/aqueduct.dart';


class PersonController extends HTTPController {

  @httpGet

  Future<Response> getAll() async {

    var query = new Query<Person>();

    

    var results = await query.fetch();

    

    return new Response.ok(results);

  }

}

Construct your data model clearly in code.

class Employee

    extends ManagedObject<_Employee>

    implements _Employee {

  bool get wasRecentlyHired {

    var now = new DateTime.now();

    return hireDate.difference(now).inDays < 30;

  }

}


class _Employee  {

  @managedPrimaryKey

  int index;


  @ManagedColumnAttributes(indexed: true)

  String name;


  DateTime hireDate;

  int salary;


  @ManagedRelationship(#employees)

  Company company;

}
